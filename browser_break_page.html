Small Parser Sketch

def normalize(value: str) -> str:
    return " ".join(value.strip().split())

def parse_line(line: str) -> tuple[str, str] | None:
    if "=" not in line:
        return None
    key, value = line.split("=", 1)
    return normalize(key), normalize(value)
